1. Get students' names who come from "Computer Science Department"
select u.user_name from users u, belongs b, department d where d.dep_name = 'Computer Science Department' and b.dep_id = d.dep_id and u.user_id = b.user_id;

2. Find the questions asked by user 'fp2360'
select q.que_title from question q inner join ask a on q.que_id = a.que_id where a.user_id = 'fp2360';

3. Get the number of assignments of each course opened in 2016 Fall(output the result at the descending order).
select o.course_id, o.session, count(*) as sum from postassignment p,open o where o.year = 2016 and o.season = 'Fall' and o.open_cid = p.open_cid group by o.course_id, o.session order by count(*) desc;
